function getNthNum(n) {
  let str = "1";
  n -= 1;
  while (n--) {
    let strNum = "";
    for (let i = 0; i < str.length; i++) {
      let count = 1;
      while (str[i] === str[i + 1] && i < str.length) {
        count += 1;
        i++;
      }
      strNum += count + str[i];
    }
    str = strNum;
  }
  return str;
}
