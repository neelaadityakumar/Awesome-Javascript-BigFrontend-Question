```js
/**
 * @param {any} data
 * @return {string}
 */
function detectType(data) {
  if (data === null) {
    return "null";
  }
  if (data === undefined) {
    return "undefined";
  }
  return data.constructor.name.toLowerCase();
}

function detectType(data) {
  if (data instanceof FileReader) return "object";
  return Object.prototype.toString
    .call(data)
    .slice(1, -1)
    .split(" ")[1]
    .toLowerCase();
}
```
